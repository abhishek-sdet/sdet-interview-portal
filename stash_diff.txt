+/**
+ * Parse answer key section
+ * Returns array of answer strings (e.g., ['A', 'C', 'B'])
+ */
+function parseAnswerKey(section) {
+    const answers = [];
+    const lines = section.content.split('\n').map(l => l.trim()).filter(l => l);
+
+    for (const line of lines) {
+        // Match "1. A" or "A. Option Text" or just "A"
+        // Priority to simple "A." or "1. A" formats
+
+        // Format: "1. A" or "1. A." or "1 A"
+        const numberedMatch = line.match(/^\d+[\.)\s]+([A-Da-d])(?:[\.)]|$)/);
+        if (numberedMatch) {
+            answers.push(numberedMatch[1].toUpperCase());
+            continue;
+        }
+
+        // Format: "A. Option Text" (used in SET A.docx for answers)
+        // Only accept if it starts with A-D and dot/paren
+        const letterMatch = line.match(/^([A-Da-d])[\.)]/);
+        if (letterMatch) {
+            answers.push(letterMatch[1].toUpperCase());
+            continue;
+        }
+
+        // Format: Just "A" or "A)" or "(A)"
+        const simpleMatch = line.match(/^[\(]?([A-Da-d])[\)]?$/);
+        if (simpleMatch) {
+            answers.push(simpleMatch[1].toUpperCase());
+            continue;
+        }
+    }
+
+    return answers;
